// always show the first match
db.students.findOne({name: "Michael"})



// update documents
db.personal.updateOne({name: "Abhishek"}, {$set: {name: "Abhishek singh"}}) // $set is operator we always prepend $ while using operators in mongoDB
db.personal.updateMany({name: "Abhishek"}, {$set: {name: "Abhishek singh"}})


// Delete Documents

db.personal.deleteOne({name: "Abhishek"}, {$set: {name: "Abhishek singh"}}) // $set is operator we always prepend $ while using operators in mongoDB
db.personal.deleteMany({name: "Abhishek"}, {$set: {name: "Abhishek singh"}})
db.personal.deleteMany({}) // it will delete all the documents in collections, hence the database will be empty
